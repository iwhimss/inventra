```

start.bat

./start.sh

# Doğrudan çalıştırma
dart run bin/server.dart --setup      # İlk kurulum
dart run bin/server.dart              # Başlat (LAN/VDS)
dart run bin/server.dart --local      # Sadece bu cihaz
dart run bin/server.dart --port 8080  # Farklı port
```
